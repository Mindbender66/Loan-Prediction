def cramers_V(v1,v2):
    crosstab = n.array(p.crosstab(v1,v2,rownames=None, colnames=None))
    stat = chi2_contingency(crosstab)[0]
    obs = n.sum(crosstab)
    mini = min(crosstab.shape)-1
    return (stat/(obs*mini))
